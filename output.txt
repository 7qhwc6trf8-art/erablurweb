TelegramButton.tsx
export function Button({ children, onClick, variant = 'primary' }: any) {
  const base =
    "w-full py-3 rounded-xl font-medium transition active:scale-95"

  const styles =
    variant === 'primary'
      ? "bg-[var(--button)] text-[var(--button-text)]"
      : "bg-white/10 text-[var(--text)]"

  return (
    <button onClick={onClick} className={`${base} ${styles}`}>
      {children}
    </button>
  )
}

export function getTheme(tg: any) {
  return {
    bg: tg.themeParams.bg_color,
    text: tg.themeParams.text_color,
    hint: tg.themeParams.hint_color,
    card: tg.themeParams.secondary_bg_color,
    button: tg.themeParams.button_color,
    buttonText: tg.themeParams.button_text_color,
  }
}
